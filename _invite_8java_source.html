136  public boolean hasAlarm()
137  {
138  if (mInvite.hasAlarm())
139  {
140  Alarm alarm = getDisplayAlarm();
141 
142  if (alarm != null)
143  {
144  Alarm.TriggerType triggerType = getTriggerType(alarm);
145  if (Alarm.TriggerType.ABSOLUTE.equals(triggerType))
146  {
147  return true;
148  }
149 
150  Alarm.TriggerRelated triggerRelated = getTriggerRelated(alarm);
151  if (Alarm.TriggerType.RELATIVE.equals(triggerType))
152  {
153  if (Alarm.TriggerRelated.START.equals(triggerRelated))
154  {
155  return hasStartTime();
156  }
157 
158  if (Alarm.TriggerRelated.END.equals(triggerRelated))
159  {
160  return hasEndDate();
161  }
162  }
163  }
164  }
165 
166  return false;
167  }
