170  private void readExtensions(File parentDirectory)
171  {
172  String files[] = parentDirectory.list();
173 
174  if (files == null || files.length == 0) {
175  return;
176  }
177 
178  LinkedList<File> libraries = new LinkedList<File>();
179  LinkedList<ExtensionInfo> extensionInfoList = new LinkedList<ExtensionInfo>();
180 
181  for (String fileName : files)
182  {
183  File file = new File(parentDirectory, fileName);
184  if (!file.isFile()) {
185  continue;
186  }
187 
188  try
189  {
190  ZipFile zipFile = new ZipFile(file);
191 
192  libraries.add(file);
193 
194  ZipEntry entry = zipFile.getEntry("META-INF/MANIFEST.MF");
195  if (entry == null) {
196  continue;
197  }
198 
199  InputStream stream = null;
200  try
201  {
202  stream = zipFile.getInputStream(entry);
203  Manifest manifest = new Manifest(stream);
204 
205  String zalExtensionClass = manifest.getMainAttributes().getValue("ZAL-Extension-Class");
206  if (zalExtensionClass == null || zalExtensionClass.isEmpty())
207  {
208  continue;
209  }
210 
211  String zalExtensionName = manifest.getMainAttributes().getValue("ZAL-Extension-Name");
212  if (zalExtensionName == null || zalExtensionName.isEmpty())
213  {
214  zalExtensionName = zalExtensionClass;
215  }
216 
217 
218  String zalRequiredVersion = manifest.getMainAttributes().getValue("ZAL-Required-Version");
219  if (zalRequiredVersion == null || zalRequiredVersion.isEmpty())
220  {
221  zalRequiredVersion = ZalVersion.current.toString();
222  }
223 
224  ExtensionInfo extensionInfo = new ExtensionInfo(
225  zalExtensionClass,
226  zalExtensionName,
227  new Version(zalRequiredVersion)
228  );
229  extensionInfoList.add(extensionInfo);
230  }
231  finally
232  {
233  if (stream != null)
234  {
235  stream.close();
236  }
237  }
238  }
239  catch (IOException e)
240  {
241  ZimbraLog.extensions.warn("Ignoring file non zip-file " + file.getAbsolutePath());
242  continue;
243  }
244 
245  if (extensionInfoList.isEmpty())
246  {
247  ZimbraLog.extensions.warn("No extension loaded from directory " + parentDirectory.getAbsolutePath());
248  return;
249  }
250  }
251 
252  for( ExtensionInfo info : extensionInfoList )
253  {
254  if( !info.isCompatible(ZalVersion.current) )
255  {
256  ZimbraLog.extensions.warn(
257  "Unable to load extension "+info.getZalExtensionName()+": it requires ZAL version "+info.getRequiredVersion().toString()+" but current version is "+ZalVersion.current.toString()
258  );
259  continue;
260  }
261 
262  try
263  {
264  Extension extension = new Extension(
265  info.getExtensionClass(),
266  libraries
267  );
268 
269  mExtensionMap.put(
270  info.getExtensionClass(),
271  extension
272  );
273  }
274  catch (ClassNotFoundException e)
275  {
276  ZimbraLog.extensions.warn("Unable to create extension "+info.getZalExtensionName(), e);
277  }
278  }
279  }
